```rust
use std::num::NonZeroUsize;
use std::sync::Arc;
use secrecy::SecretString;
use tokio::sync::Mutex;

pub struct OutputCache(lru::LruCache<String, Arc<RdfGraph>>);

impl OutputCache {
    pub fn new(cap: usize) -> Self {
        Self(lru::LruCache::new(NonZeroUsize::new(cap).unwrap()))
    }
    pub fn get(&mut self, key: &str) -> Option<Arc<RdfGraph>> {
        self.0.get(key).cloned()
    }
    pub fn insert(&mut self, key: String, graph: RdfGraph) -> Arc<RdfGraph> {
        let arc = Arc::new(graph);
        self.0.put(key, arc.clone());
        arc
    }
    pub fn remove(&mut self, key: &str) {
        self.0.pop(key);
    }
}

pub struct AppState {
    pub db: Database,
    pub runner: Arc<JobRunner>,
    pub catalog: Arc<RdfGraph>,
    pub output_cache: Arc<Mutex<OutputCache>>,
    pub api_key: SecretString,
    pub rate_limiter: RateLimiter,
}
```

```rust
//! Security smoke test: asserts all non-public API endpoints return 401
//! for requests with no valid session cookie.

//! This test satisfies ROADMAP Phase 03 Success Criterion #4:

//! session cookie; a security smoke test asserts this for all routes"

use axum::http::{Request, StatusCode};
use tower::ServiceExt; // for oneshot

/// Helper: build a test router backed by a temp directory SQLite DB.
async fn test_router() -> axum::Router {
    // 1. Create a tempdir for the test database
    // 2. Database::open(temp_path, None)
    // 3. Create a session store with a separate tokio_rusqlite connection to same DB
    // 4. Migrate session store
    // 5. Construct minimal AppState (JobRunner, RdfGraph, OutputCache, rate_limiter, api_key)
    // 6. Call build_router(state, None, session_store, session_secret)
    // 7. Return the Router
}

/// All protected routes must return 401 without a session.
#[tokio::test]
async fn protected_routes_return_401_without_session() {
    let app = test_router().await;

    // List of (method, path) tuples for ALL protected routes.
    // Includes a representative with path params substituted to a dummy UUID.
    let dummy_id = "00000000-0000-0000-0000-000000000099";
    let protected = vec![
        ("POST", "/v1/auth/logout"),
        ("GET",  "/v1/jobs"),
        ("POST", "/v1/jobs"),
        ("GET",  format!("/v1/jobs/{dummy_id}").as_str()),
        // ... ALL protected routes from the interfaces block above
    ];

    for (method, path) in &protected {
        let req = Request::builder()
            .method(*method)
            .uri(*path)
            .body(axum::body::Body::empty())
            .unwrap();

        let response = app.clone().oneshot(req).await.unwrap();
        assert_eq!(
            response.status(),
            StatusCode::UNAUTHORIZED,
            "{method} {path} should return 401 without session, got {}",
            response.status()
        );
    }
}

/// Public routes must NOT return 401 without a session.
#[tokio::test]
async fn public_routes_do_not_return_401() {
    let app = test_router().await;

    let public = vec![
        ("GET",  "/healthz/live"),
        ("GET",  "/healthz/ready"),
        ("GET",  "/v1/settings/schema"),
        ("GET",  "/v1/providers"),
        ("POST", "/v1/auth/register"),
        ("POST", "/v1/auth/login"),
    ];

    for (method, path) in &public {
        let req = Request::builder()
            .method(*method)
            .uri(*path)
            .body(axum::body::Body::empty())
            .unwrap();

        let response = app.clone().oneshot(req).await.unwrap();
        assert_ne!(
            response.status(),
            StatusCode::UNAUTHORIZED,
            "{method} {path} should NOT return 401 (public route), got 401",
        );
    }
}
```
